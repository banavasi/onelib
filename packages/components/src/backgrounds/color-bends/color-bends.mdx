## Usage

Copy the component into your project at `src/components/backgrounds/color-bends/color-bends.tsx`:

```tsx
// Install peer dependency: pnpm add three
import ColorBends from "@/components/backgrounds/color-bends";

export default function HeroSection() {
  return (
    <div className="relative w-full h-screen">
      <ColorBends
        rotation={45}
        speed={0.2}
        colors={["#FF6B35", "#00BFFF", "#7CFF67"]}
        warpStrength={1}
      />
    </div>
  );
}
```
