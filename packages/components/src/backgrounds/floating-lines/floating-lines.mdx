## Usage

Copy the component into your project at `src/components/backgrounds/floating-lines/floating-lines.tsx`:

```tsx
// Install peer dependency: pnpm add three
import FloatingLines from "@/components/backgrounds/floating-lines";

export default function HeroSection() {
  return (
    <div className="relative w-full h-screen">
      <FloatingLines
        linesGradient={["#FF9FFC", "#5227FF", "#7CFF67"]}
        animationSpeed={1}
        interactive={true}
        parallax={true}
      />
    </div>
  );
}
```
