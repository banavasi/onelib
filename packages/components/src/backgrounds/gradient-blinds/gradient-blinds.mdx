## Usage

Copy the component into your project at `src/components/backgrounds/gradient-blinds/gradient-blinds.tsx`:

```tsx
// Install peer dependency: pnpm add ogl
import GradientBlinds from "@/components/backgrounds/gradient-blinds";

export default function HeroSection() {
  return (
    <div className="relative w-full h-screen">
      <GradientBlinds
        gradientColors={["#FF9FFC", "#5227FF"]}
        blindCount={16}
        noise={0.3}
      />
    </div>
  );
}
```
