## Usage

Copy the component into your project at `src/components/backgrounds/light-rays/light-rays.tsx`:

```tsx
// Install peer dependency: pnpm add ogl
import LightRays from "@/components/backgrounds/light-rays";

export default function HeroSection() {
  return (
    <div className="relative w-full h-screen">
      <LightRays
        raysOrigin="top-center"
        raysColor="#ffffff"
        raysSpeed={1}
        lightSpread={1}
        rayLength={2}
      />
    </div>
  );
}
```
