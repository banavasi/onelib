# Tabs

Animated tab component with a spring-based active indicator and smooth content transitions using Motion. Accepts an array of tab items with label, value, and React node content.

- **Source:** [Sera UI](https://seraui.com)
- **Category:** Sections
- **Peer Dependencies:** `motion`

## Usage

```tsx
import { Tabs } from "@/components/sections/tabs";
import type { TabItem } from "@/components/sections/tabs/tabs";

const items: TabItem[] = [
  {
    label: "Overview",
    value: "overview",
    content: <div>Overview content</div>,
  },
  {
    label: "Details",
    value: "details",
    content: <div>Details content</div>,
  },
];

export default function Example() {
  return <Tabs items={items} defaultValue="overview" />;
}
```
