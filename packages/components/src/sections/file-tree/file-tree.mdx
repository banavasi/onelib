## Usage

```tsx
import { FileTree } from "@/components/sections/file-tree";
import type { FileTreeItem } from "@/components/sections/file-tree";

const myFiles: FileTreeItem[] = [
  {
    name: "src",
    type: "folder",
    children: [
      { name: "index.ts", type: "file" },
      { name: "App.tsx", type: "file" },
    ],
  },
  { name: "package.json", type: "file" },
];

export default function Sidebar() {
  return <FileTree data={myFiles} />;
}
```
