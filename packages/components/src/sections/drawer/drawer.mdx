# Drawer

A drawer component that slides in from any side of the screen with smooth spring-based animations. Built with Motion for fluid transitions.

- **Source:** [Sera UI](https://seraui.com)
- **Category:** Sections
- **Peer Dependencies:** `motion`

## Usage

```tsx
import {
	Drawer,
	DrawerOverlay,
	DrawerContent,
	DrawerHeader,
	DrawerTitle,
	DrawerDescription,
	DrawerFooter,
} from "@/components/sections/drawer";
import { useState } from "react";

export default function Example() {
	const [isOpen, setIsOpen] = useState(false);

	return (
		<>
			<button onClick={() => setIsOpen(true)}>Open Drawer</button>
			<Drawer open={isOpen} onOpenChange={setIsOpen} side="right">
				<DrawerOverlay />
				<DrawerContent>
					<DrawerHeader>
						<DrawerTitle>Title</DrawerTitle>
						<DrawerDescription>Description text</DrawerDescription>
					</DrawerHeader>
					<div className="p-6">Content here</div>
					<DrawerFooter>
						<button onClick={() => setIsOpen(false)}>Close</button>
					</DrawerFooter>
				</DrawerContent>
			</Drawer>
		</>
	);
}
```
